class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n = nums.size();
        if(n < 1) return 0; 
        int slow = 1, fast = 1;
        while(fast < n){
            if(nums[fast]!= nums[fast-1]){
            nums[slow] = nums[fast];
            slow++;
            }
            ++fast;
        }
        return slow;
    }
};
